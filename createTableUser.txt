##创建用户表
create table t_user (
	user_id int auto_increment primary key,  ##用户ID
	user_name VARCHAR(30),  ##用户名
	password VARCHAR(32),  ##密码
	email varchar(50),  ##邮箱
	
	last_visit datetime,  ##最后登录时间
	last_ip varchar(23),  ##最后登录IP
	
	create_date datetime not null, ##创建时间
	create_by varchar(30) not null, ##创建人
	update_date datetime not null, ##更新时间
	update_by varchar(30) not null ##更新人
)ENGINE=InnoDB;

##创建用户登录日志表
create table t_login_log (
	login_log_id int auto_increment primary key, ##日志ID
	user_id int not null, ##用户ID
	user_ip varchar(23), ##用户登录IP
	create_date datetime not null ##创建时间
)ENGINE=InnoDB;
